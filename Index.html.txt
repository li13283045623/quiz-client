    const adminBanks = ref([]);
    const adminUsers = ref([]);
    const adminSelfId = computed(() => currentUser.value?._id || '');
    const uploadForm = ref({ bankName: '', text: '' });
    const uploading = ref(false);
    const uploadMsg = ref('');

    // 工具类：网络高层封装
    const api = async (url, options = {}) => {
      options.headers = options.headers || {};
      if (token.value) {
        options.headers['Authorization'] = 'Bearer ' + token.value;
      }
      if (options.body && !(options.body instanceof FormData)) {
        options.headers['Content-Type'] = 'application/json';
      }
      const res = await fetch(API + url, options);
      if (!res.ok) {
        const errData = await res.json().catch(() => ({}));
        throw new Error(errData.message || '网络底端握手失败');
      }
      return res.json();
    };

    // 鉴权管道
    const doAuth = async () => {
      authLoading.value = true;
      authError.value = '';
      try {
        const path = authTab.value === 'login' ? '/auth/login' : '/auth/register';
        const data = await api(path, { method: 'POST', body: JSON.stringify(authForm.value) });
        token.value = data.token;
        currentUser.value = data.user;
        localStorage.setItem('quiz_token', data.token);
        localStorage.setItem('quiz_user', JSON.stringify(data.user));
        initApp();
      } catch (err) {
        authError.value = err.message;
      } finally {
        authLoading.value = false;
      }
    };

    const logout = () => {
      token.value = '';
      currentUser.value = null;
      localStorage.removeItem('quiz_token');
      localStorage.removeItem('quiz_user');
      page.value = 'home';
    };

    // 核心题库抓取调度
    const fetchBanks = async () => {
      try { banks.value = await api('/questions/banks'); } catch(e){}
    };

    const openBank = async (bank) => {
      currentBank.value = bank;
      try {
        currentBankQuestions.value = await api('/questions/bank/' + bank._id);
        page.value = 'bank-detail';
      } catch (err) { alert('加载题库失败'); }
    };

    const goHome = () => {
      page.value = 'home';
      fetchBanks();
    };

    // 刷题引擎计算属性绑定
    const currentPQ = computed(() => practiceQuestions.value[practiceStep.value] || {});
    const isFavorited = (qid) => favoriteList.value.some(f => f._id === qid);
    const accuracy = computed(() => {
      if (!meData.value.history || meData.value.history.length === 0) return 0;
      const correct = meData.value.history.filter(h => h.isCorrect).length;
      return Math.round((correct / meData.value.history.length) * 100);
    });

    // 练习核心方法簇
    const startPractice = () => {
      if (currentBankQuestions.value.length === 0) return alert('该库下没有任何可供练习的题目！');
      practiceMode.value = 'normal';
      practiceQuestions.value = [...currentBankQuestions.value];
      resetPracticeSession();
    };

    const resetPracticeSession = () => {
      practiceStep.value = 0;
      practiceAnswered.value = false;
      practiceSelectedOption.value = null;
      page.value = 'practice';
    };

    const selectOption = async (idx) => {
      if (practiceAnswered.value) return;
      practiceSelectedOption.value = idx;
      practiceAnswered.value = true;
      const q = currentPQ.value;
      const isRight = idx === q.correctAnswer;
      try {
        await api('/questions/submit', { method: 'POST', body: JSON.stringify({ questionId: q._id, selectedAnswer: idx, isCorrect: isRight }) });
      } catch(e){}
    };

    const getPracticeOptionClass = (idx) => {
      if (!practiceAnswered.value) {
        return practiceSelectedOption.value === idx ? 'border-indigo-500 bg-indigo-50/40 text-indigo-900 font-medium' : 'border-slate-200 hover:bg-slate-50';
      }
      if (idx === currentPQ.value.correctAnswer) return 'border-emerald-500 bg-emerald-50 text-emerald-900 font-semibold shadow-sm shadow-emerald-100';
      if (practiceSelectedOption.value === idx) return 'border-rose-500 bg-rose-50 text-rose-900';
      return 'border-slate-100 opacity-60';
    };

    const nextQuestion = () => {
      if (practiceStep.value < practiceQuestions.value.length - 1) {
        practiceStep.value++;
        practiceAnswered.value = false;
        practiceSelectedOption.value = null;
      } else {
        alert('恭喜您已刷完本卷全部题目！');
        exitPractice();
      }
    };

    const prevQuestion = () => {
      if (practiceStep.value > 0) {
        practiceStep.value--;
        practiceAnswered.value = true;
        practiceSelectedOption.value = null;
      }
    };

    const exitPractice = () => {
      page.value = currentBank.value ? 'bank-detail' : 'me';
      loadMe();
    };

    // 错题夹与收藏本数据加载
    const loadMe = async () => {
      page.value = 'me';
      try {
        meData.value = await api('/users/me');
        wrongList.value = await api('/questions/wrong');
        favoriteList.value = await api('/questions/favorites');
      } catch(e){}
    };

    const startWrongPractice = () => {
      if (wrongList.value.length === 0) return alert('恭喜你，目前没有任何错题记录！');
      practiceMode.value = 'wrong';
      practiceQuestions.value = [...wrongList.value];
      resetPracticeSession();
    };

    const startFavPractice = () => {
      if (favoriteList.value.length === 0) return alert('目前收藏夹空空如也哦！');
      practiceMode.value = 'fav';
      practiceQuestions.value = [...favoriteList.value];
      resetPracticeSession();
    };

    const toggleFavorite = async (q) => {
      const fav = isFavorited(q._id);
      try {
        if (fav) {
          await api('/questions/favorite/' + q._id, { method: 'DELETE' });
          favoriteList.value = favoriteList.value.filter(f => f._id !== q._id);
        } else {
          await api('/questions/favorite', { method: 'POST', body: JSON.stringify({ questionId: q._id }) });
          favoriteList.value.push(q);
        }
      } catch(e){}
    };

    // 考试引擎核心实现
    const startExam = () => {
      if (currentBankQuestions.value.length === 0) return alert('无可调配试题');
      const shuffled = [...currentBankQuestions.value].sort(() => 0.5 - Math.random());
      examQuestions.value = shuffled.slice(0, 20);
      examAnswers.value = {};
      examStep.value = 0;
      page.value = 'exam';
    };

    const examSelectOption = (idx) => { examAnswers.value[examStep.value] = idx; };
    
    const submitExam = () => {
      let score = 0;
      let correct = 0;
      examQuestions.value.forEach((q, i) => {
        if (examAnswers.value[i] === q.correctAnswer) correct++;
      });
      examCorrectCount.value = correct;
      examScore.value = Math.round((correct / examQuestions.value.length) * 100);
      page.value = 'exam-result';
    };

    const examScoreEmoji = computed(() => {
      if (examScore.value >= 90) return '👑';
      if (examScore.value >= 60) return '🎉';
      return '👻';
    });

    // 超级管理端逻辑维护
    const loadAdminBanks = async () => { try { adminBanks.value = await api('/questions/banks'); } catch(e){} };
    const loadAdminUsers = async () => { try { adminUsers.value = await api('/admin/users'); } catch(e){} };
    
    const deleteBank = async (id) => {
      if (!confirm('严重警告：注销题库将不可逆，确认操作？')) return;
      await api('/questions/bank/' + id, { method: 'DELETE' });
      loadAdminBanks(); fetchBanks();
    };
    
    const toggleRole = async (user) => {
      const newRole = user.role === 'admin' ? 'user' : 'admin';
      await api('/admin/users/' + user._id + '/role', { method: 'PUT', body: JSON.stringify({ role: newRole }) });
      loadAdminUsers();
    };

    const submitUpload = async () => {
      if (!uploadForm.value.bankName || !uploadForm.value.text) return alert('请确保题库名字和TXT内容全部填写完整！');
      uploading.value = true; uploadMsg.value = '';
      try {
        const res = await api('/questions/upload', { method: 'POST', body: JSON.stringify(uploadForm.value) });
        uploadMsg.value = `录入大获成功！共识别并发布题库：${res.bank?.name || ''}`;
        uploadForm.value = { bankName: '', text: '' };
        loadAdminBanks(); fetchBanks();
      } catch (err) { uploadMsg.value = '智能解析发生阻碍: ' + err.message; }
      finally { uploading.value = false; }
    };

    // 应用自发启动挂载载入
    const initApp = () => {
      fetchBanks();
      if (currentUser.value?.role === 'admin') {
        loadAdminBanks(); loadAdminUsers();
      }
    };

    onMounted(() => { if (token.value) initApp(); });

    return {
      token, currentUser, authTab, authForm, authError, authLoading, doAuth, logout,
      page, banks, meData, wrongList, favoriteList, accuracy, loadMe, goHome,
      currentBank, currentBankQuestions, openBank,
      practiceMode, practiceQuestions, practiceStep, practiceAnswered, practiceSelectedOption,
      currentPQ, isFavorited, startPractice, startWrongPractice, startFavPractice,
      selectOption, getPracticeOptionClass, nextQuestion, prevQuestion, exitPractice, toggleFavorite,
      startExam, examQuestions, examAnswers, examStep, examSelectOption, submitExam, examScore, examCorrectCount, examScoreEmoji,
      adminTab, adminBanks, adminUsers, adminSelfId, loadAdminBanks, loadAdminUsers, deleteBank, toggleRole,
      uploadForm, uploading, uploadMsg, submitUpload
    };
  }
}).mount('#app');
</script>
</body>
